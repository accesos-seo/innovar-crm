### Paso 2 — Agente aplica trigger proyecto_completado vía Management API

Verificar función enqueue:
  SELECT proname FROM pg_proc WHERE proname ILIKE 'fn_wa_enqueue%';

Luego crear función + trigger (ajustar nombre de fn_wa_enqueue según resultado):

  CREATE OR REPLACE FUNCTION notify_proyecto_completado()
  RETURNS trigger LANGUAGE plpgsql SECURITY DEFINER AS $$
  DECLARE
    v_client     RECORD;
    v_first_name text;
  BEGIN
    IF OLD.delivered_at IS NOT NULL OR NEW.delivered_at IS NULL THEN RETURN NEW; END IF;
    IF NOT COALESCE(NEW.is_fully_paid, false) THEN RETURN NEW; END IF;
    SELECT c.id, c.name INTO v_client FROM public.clients c WHERE c.id = NEW.client_id;
    IF NOT FOUND THEN RETURN NEW; END IF;
    v_first_name := split_part(COALESCE(v_client.name, 'Cliente'), ' ', 1);
    PERFORM public.fn_wa_enqueue(
      NEW.client_id, 'project.completed', 'proyecto_entregado', 'project', NEW.id,
      'proyecto_completado_v1',
      jsonb_build_array(v_first_name, COALESCE(NEW.name, 'tu proyecto')),
      jsonb_build_object('project_id', NEW.id, 'delivered_at', NEW.delivered_at)
    );
    RETURN NEW;
  END;
  $$;

  DROP TRIGGER IF EXISTS trg_notify_proyecto_completado ON public.projects;
  CREATE TRIGGER trg_notify_proyecto_completado
    AFTER UPDATE OF delivered_at ON public.projects
    FOR EACH ROW EXECUTE FUNCTION notify_proyecto_completado();
